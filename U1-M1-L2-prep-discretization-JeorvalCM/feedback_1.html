def disc_eq_freq(X, k):
    # Implement equal frequency discretization with k bins.
    # Each feature should be discretized individually (univariate discretization).
    # Your code should work with a data matrix X with an arbitrary number of unsorted features.
    
    m, n = X.shape
    
    # Find class limits.
    # Hint: use -inf and +inf as the first and last limit
    # YOUR CODE HERE
    percentiles = np.arange(100/k,100,100/k)
    clims = np.percentile(X, percentiles, axis = 0).T
    m1, n1 = clims.shape
    clims = np.insert(clims, n1, np.inf, axis = 1)
    clims = np.insert(clims, 0, np.NINF, axis = 1)
    
    
    X_disc = np.zeros(X.shape)
    for i in range(n):
        # YOUR CODE HERE
        X_disc[:,i] = np.searchsorted(clims[i], X[:,i], side='right') - 1
        
    # Return the discretized labels in a matrix the same shape as X.
    # Return the limits in a matrix which rows are the interval classes of each feature.
    return X_disc, clims

def k_means_1D(x, k):
    """ Implmentes K-means discretization.
    
    INPUT:
        x: one dimensional numpy array with the features to discretize
        k: number of bins
        
    OUTPUT:
        b: the bin edges
    """
    n = len(x)
    x = np.array(x)
    
    # We need an array for the cluster index of each point
    cluster = np.zeros(n, dtype=int)
    
    # Bin edges
    b = np.zeros(k + 1)
    
    # This variable monitors the reallocation of clusters.
    # change = true if at least one value is being moved to a different cluster.
    change = False
    
    # Initialize cluster centers to first unique k elements in x
    C = []
    for e in x:
        if e in C:
            continue
        C.append(e)
        if len(C) == k:
            break
    C = np.sort(C)
    if len(C) < k:
        assert "Error, not enough unique centers."
        
    # initialize bin edges
    b[0] = -np.inf
    b[-1] = np.inf
    b[1:-1] = (C[1:] + C[:-1])/2
    
    # Assign cluster index in cluster list
    # YOUR CODE HERE
    for i in range(n):
        

        #getting the closest center by applying a substraction, so the minimum value is the closest center
        absolute_val_array = np.abs(C - x[i])

        #getting the index of the closest center
        closest_center = absolute_val_array.argmin()

        #saving the index of the closest center, in the cluster list
        cluster[i] = closest_center
            
        """for j in range(k+1):
            if x[i] >= b[j] and x[i] < b[j+1]:
                cluster[i] = j
                break"""
        
    
    # Iteration
    change = True
    while change:
        change = False
        # Recompute the centers of the clusters as 
        # the average of the values in each cluster.
        
        #part to compute the new center
        cluster_values = [[] for _ in range(len(C))]
        for i in range(n):
            
            cluster_values[cluster[i]].append(x[i])
            
        #the reason why I obtained the mean in this way, was because since each cluster has different lenghts,the numpy array was not a matrix
        #but a list so applying mean to a list raises an error.
        C = np.array([np.mean(values) for values in cluster_values])
        
        
        
        # YOUR CODE HERE
        #part obtained in the pseudocode to obtain the new group of each point if possible
        for i in range(n):
            
            #obtaining current index of the x on its group
            j = cluster[i]
            
            #the if is in case it is not possible j - 1, case when j is 0
            if j > 0:
                if (x[i] < C[j]) and ((x[i]-C[j-1]) < (C[j] - x[i])):
                    cluster[i] = j - 1
                    change = True
                    
            else:
                pass
                
            #the if is in case it is not possible j + 1, case when j is k - 1
            if j < k-1:
                if (x[i] > C[j]) and ((C[j+1] - x[i]) < (x[i] - C[j])):
                    cluster[i] = j + 1
                    change = True
            else:
                pass
        
    # Determination of the cut points
    b[0] = x.min()
    b[-1] = x.max()
    b[1:-1] = (C[1:] + C[:-1])/2
    
    return b

def disc_k_means(X, k):
    """ Implmentes K-means discretization.
    
    INPUT:
        X: data matrix as a numpy array
        k: number of bins
    """
    m, n = X.shape
    X_disc = np.zeros(X.shape)
    clims = []
    
    for i in range(n):
        bin_edges = k_means_1D(X[:,i], k)
        bin_edges[0] = -np.inf
        bin_edges[-1] = np.inf
        clims.append(list(bin_edges))
        X_disc[:,i] = np.searchsorted(bin_edges, X[:,i], side='right') - 1
        
    return X_disc, np.array(clims)

from sklearn.model_selection import KFold
from sklearn.model_selection import GridSearchCV
from sklearn.model_selection import cross_val_score


def disc_kde_1D(x, plot=False):
    """ Implmentes KDE discretization.
    
    INPUT:
        x: one dimensional numpy array with the features to discretize
               
    OUTPUT:
        b: the bin edges
    """
    
    n = len(x)
    x = np.array(x)
    x.sort()
    
    # Select an appropiate bw candidates
    xmin = x[0]
    xmax = x[-1]
    xrange = xmax - xmin
    bw_l = np.linspace(0.01*xrange, xrange, 100)
    bw_scores = np.zeros(len(bw_l))
    
    # Use 10-fold CV. Find the corresponding train and test for each fold.
    # Note: Since x is ordered, maybe systematic sampling would give better cv results.
    
    #Note: seems that using gridsearchCV sometimes fails
    #parameters = [{'bandwidth': bw_l}]
    #kde = KernelDensity(kernel='gaussian')
    #kde_grid = GridSearchCV(kde, param_grid = parameters, cv = 10, n_jobs = -1)
    #kde_grid.fit(x.reshape((-1,1)))
    #best_bw = kde_grid.best_params_['bandwidth']
    #random_state is 11 in order to have a similar graph to the notebook class
    folds = list(KFold(n_splits=10, shuffle=True, random_state = 11).split(x)) 
    for i, bw in enumerate(bw_l):
        # YOUR CODE HERE
        
        kde = KernelDensity(kernel='gaussian', bandwidth=bw)
        
        bw_scores[i] = np.mean(cross_val_score(kde, X = x.reshape(-1,1) , cv = folds))
        
    # It seems the best bw benefits from some smoothing, need to explore this more, systematic sampling may solve this.
    best_bw = 1.5*bw_l[np.argmax(bw_scores)]
    #best_bw = 1.5*best_bw
    
    # List to store bin edges
    bin_edges = [xmin]
    
    # Find local minima using a grid, other option is to use data points.
    grid_x = np.linspace(xmin, xmax, 100)
    kde_full = KernelDensity(bandwidth=best_bw, kernel='gaussian').fit(x.reshape(-1,1)).score_samples(grid_x.reshape(-1,1))
    
    for i in range(1, len(kde_full) - 1):
        if (kde_full[i-1] > kde_full[i]) and (kde_full[i+1] > kde_full[i]): bin_edges.append(grid_x[i])
    
    bin_edges.append(xmax)
    
    return np.array(bin_edges)
