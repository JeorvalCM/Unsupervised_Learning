# Build table with label count
# each row is a cluster
dict_to_pd = {key:0 for key in data['target_names']}

dict_clusters_single = {key: dict_to_pd.copy() for key in range(1, 4)}

dict_clusters_complete = {key: dict_to_pd.copy() for key in range(1, 4)}

dict_clusters_average = {key: dict_to_pd.copy() for key in range(1, 4)}

dict_clusters_weighted = {key: dict_to_pd.copy() for key in range(1, 4)}

dict_clusters_ward = {key: dict_to_pd.copy() for key in range(1, 4)}

for idx, val in enumerate(Y):
    
    dict_clusters_single[clusters_labels_single[idx]][val] += 1
    
    dict_clusters_complete[clusters_labels_complete[idx]][val] += 1
    
    dict_clusters_average[clusters_labels_average[idx]][val] += 1
    
    dict_clusters_weighted[clusters_labels_weighted[idx]][val] += 1
    
    dict_clusters_ward[clusters_labels_ward[idx]][val] += 1

single = pd.DataFrame(dict_clusters_single).T

complete = pd.DataFrame(dict_clusters_complete).T

average = pd.DataFrame(dict_clusters_average).T

weighted = pd.DataFrame(dict_clusters_weighted).T

ward = pd.DataFrame(dict_clusters_ward).T
