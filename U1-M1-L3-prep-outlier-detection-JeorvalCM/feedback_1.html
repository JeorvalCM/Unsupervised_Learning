def distance_matrix(X):
    """Returns the distance matrix from a data matrix X. """
    
    n, m = X.shape
    
    D = np.zeros((n ,n))
    
    # YOUR CODE HERE
    for idx, row in enumerate(X):
        for idx_2, row_ in enumerate(X):
            D[idx, idx_2] = EuclideanDistance(row, row_)
        
    return D

def EuclideanDistance(Xrow1, Xrow2):       
        """
        Function to compute the Euclidean Distance
        
        Input:
        Xrow1, Xrow1: lists containing the values to compute the distance
        
        Output:
        return the Euclidean distance 
        """
        Distance = np.sqrt(sum(np.square(Xrow1 - Xrow2)))
        
        return Distance

def scores_knn(X, k, kth=False):
    """ Return the oulier scores given by the distance to the kth neighbor."""
    
    # Find the distance matrix
    # YOUR CODE HERE
    D = distance_matrix(X)
    
    # Sort the distance matrix row by row.
    # YOUR CODE HERE
    D = np.sort(D, axis = 1)
    
    if kth:
        # Take the k column (excluding self distance)
        # YOUR CODE HERE
        scores = D[:,k]
        
    else:
        # Take the average up to the k column (excluding self distance)
        # YOUR CODE HERE
        scores = np.mean(D[:, 1:k+1], axis = 1)
    
    return scores

def lof(X, k):
    # Find the distance matrix
    # YOUR CODE HERE
    D = distance_matrix(X)
    
    #copy to later use
    D2 = np.copy(D)

    # Sort the distance matrix row by row to obtain the k neighborhood of each row.
    # YOUR CODE HERE
    D = np.sort(D, axis = 1)

    # Store the k-distance of each observation
    # YOUR CODE HERE
    #we kind the k-distance by getting the column k of the sorted row by row matrix
    k_dist = D[:,k]

    # Also store the indices of the neighbors to find the k-distance of each point o.
    # YOUR CODE HERE
    #to get the indixes we get the indixes of the values respectively sorted
    Nk_idx = np.argsort(D2, axis = 1)
    
    #eliminating the values of the self distance
    Nk_idx = np.array([np.delete(i, np.where(i == idx)) for idx, i in enumerate(Nk_idx)])[:,:k]
    

    # Find the reachability distances of each neighborhood
    # Note: Numpy fancy indexing is your friend
    r_dist = np.zeros((len(X), k))
    for i, x in enumerate(X):
        for j, o in enumerate(Nk_idx[i]):
            r_dist[i,j] = max(D2[i,o], k_dist[o])

    # YOUR CODE HERE

    # Find LDR for each observation
    # YOUR CODE HERE
    #we copute the mean of each row and then is elevated to -1
    LDR = 1/np.mean(r_dist, axis = 1)

    #Find LOF scores
    # YOUR CODE HERE
    LOF = np.zeros(X.shape[0])
    
    #we get the LOF of each row by adding all the values of all the LDR divided by LDR[i] and them divided by k
    for i, idx in enumerate(Nk_idx):
        LOF[i] = np.sum(LDR[idx]/LDR[i])/k

    scores = LOF
    return scores

def abof(a, X):
    """ Returns abof score for X[a] """

    m = len(X)
    # Variablea to store the first and second terms of the variance
    var1 = 0
    var2 = 0
    # variable to store the normalization constant (sum of weights)
    norm_c = 0

    # Loop over all pairs of points
    for b in range(m):
        if a == b or np.array_equal(X[a],X[b]):
            continue
        for c in range(b+1, m):
            if a == c or np.array_equal(X[a],X[c]):
                continue
            # YOUR CODE HERE
            subs_ba = X[b] - X[a]
            
            subs_ca = X[c] - X[a]
            
            #getting the norm of the vector
            n1 = np.linalg.norm(subs_ba)
            
            n2 = np.linalg.norm(subs_ca)
            
            #getting the scalar product, esto cause que de un resultado ligerament diferente 
            scalar_product = np.dot(subs_ba, subs_ca)
            
            var1 += np.square(scalar_product / (np.square(n1) * np.square(n2) * n1 * n2))
            var2 += (scalar_product / (np.square(n1) * np.square(n2) * n1 * n2))
            
            
            norm_c += 1/(n1 * n2)
 
    var = var1/norm_c - (var2/norm_c)**2

    return var

def abod(X):
    """ Retrun abof scores for X """
    
    scores = []

        scores.append(abof(a, X))

    return np.array(scores)
